    date_str = row["event_time"].strftime("%Y-%m-%d")
    if date_str not in embeddings_by_date:
        embeddings_by_date[date_str] = []
    embeddings_by_date[date_str].append(embeddings[idx])

    embeddings_by_date[date_str] = torch.stack(embeddings_by_date[date_str])

    "hidden_size": model.config.hidden_size,
    "model_id": MODEL_ID,
    "n_events": len(embeddings),
